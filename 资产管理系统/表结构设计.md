 1. asset_account - 用户资产总览表 (满足基本要求)

  CREATE TABLE `asset_account` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `total_balance` DECIMAL(12,2) NOT NULL DEFAULT 0.00 COMMENT '总现金余额',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_user_id` (`user_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户资产总览表';

  2. asset_income - 收入流水表 (月薪+突发收入)

  CREATE TABLE `asset_income` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `income_id` VARCHAR(32) NOT NULL COMMENT '收入ID',
    `income_type` VARCHAR(32) NOT NULL COMMENT '收入类型; MONTHLY_SALARY-月薪、BONUS-奖金、STOCK-股票、OTHER-其他',
    `amount` DECIMAL(12,2) NOT NULL COMMENT '收入金额',
    `income_date` DATE NOT NULL COMMENT '收入日期',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_income_id` (`income_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='收入流水表';

  3. asset_debt - 欠款记录表 (借贷+突发欠款)

  CREATE TABLE `asset_debt` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `debt_id` VARCHAR(32) NOT NULL COMMENT '欠款ID',
    `debt_type` VARCHAR(32) NOT NULL COMMENT '欠款类型; LOAN-贷款、CREDIT_CARD-信用卡、PERSONAL-个人借贷、OTHER-其他',
    `debt_name` VARCHAR(128) NOT NULL COMMENT '欠款名称',
    `total_amount` DECIMAL(12,2) NOT NULL COMMENT '欠款总额',
    `remaining_amount` DECIMAL(12,2) NOT NULL COMMENT '剩余欠款金额',
    `interest_rate` DECIMAL(5,4) DEFAULT NULL COMMENT '年利率(0.05表示5%)',
    `start_date` DATE NOT NULL COMMENT '欠款开始日期',
    `end_date` DATE DEFAULT NULL COMMENT '预计还清日期',
    `monthly_payment` DECIMAL(12,2) DEFAULT NULL COMMENT '每月还款金额(固定还款)',
    `is_paid_off` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '是否已还清',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_debt_id` (`debt_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='欠款记录表';

  4. asset_expense - 消费流水表 (固定消费+一次性消费)

  CREATE TABLE `asset_expense` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `expense_id` VARCHAR(32) NOT NULL COMMENT '消费ID',
    `expense_type` VARCHAR(32) NOT NULL COMMENT '消费类型; RENT-房租、MORTGAGE-房贷、FOOD-饮食、TRANSPORT-交通、OTHER-其他',
    `expense_name` VARCHAR(128) NOT NULL COMMENT '消费名称',
    `amount` DECIMAL(12,2) NOT NULL COMMENT '消费金额',
    `expense_date` DATE NOT NULL COMMENT '消费日期',
    `is_recurring` BOOLEAN NOT NULL DEFAULT FALSE COMMENT '是否为周期性消费',
    `recurring_period` VARCHAR(32) DEFAULT NULL COMMENT '周期; WEEKLY-周、MONTHLY-月、QUARTERLY-季度、YEARLY-年',
    `end_date` DATE DEFAULT NULL COMMENT '周期结束日期(固定消费的截止日期)',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_expense_id` (`expense_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='消费流水表';

  5. asset_savings_plan - 存款计划表

  CREATE TABLE `asset_savings_plan` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `plan_id` VARCHAR(32) NOT NULL COMMENT '计划ID',
    `plan_name` VARCHAR(128) NOT NULL COMMENT '计划名称',
    `current_amount` DECIMAL(12,2) NOT NULL DEFAULT 0.00 COMMENT '当前已存金额',
    `monthly_deposit` DECIMAL(12,2) NOT NULL COMMENT '每月存入金额',
    `interest_rate` DECIMAL(5,4) NOT NULL COMMENT '年利率',
    `start_date` DATE NOT NULL COMMENT '开始日期',
    `target_amount` DECIMAL(12,2) DEFAULT NULL COMMENT '目标金额(可选)',
    `target_date` DATE DEFAULT NULL COMMENT '目标日期(可选)',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_plan_id` (`plan_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='存款计划表';

  6. asset_snapshot - 资产快照表 (用于趋势分析)

  CREATE TABLE `asset_snapshot` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `snapshot_date` DATE NOT NULL COMMENT '快照日期',
    `total_assets` DECIMAL(12,2) NOT NULL COMMENT '总资产(余额+存款)',
    `total_debt` DECIMAL(12,2) NOT NULL COMMENT '总负债',
    `net_assets` DECIMAL(12,2) NOT NULL COMMENT '净资产',
    `total_income` DECIMAL(12,2) NOT NULL COMMENT '当月总收入',
    `total_expense` DECIMAL(12,2) NOT NULL COMMENT '当月总支出',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    PRIMARY KEY (`id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='资产快照表';
