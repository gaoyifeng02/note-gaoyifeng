 1. asset_account - 用户资产总览表 (满足基本要求)

  CREATE TABLE `asset_account` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `total_balance` DECIMAL(12,2) NOT NULL DEFAULT 0.00 COMMENT '总现金余额',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_user_id` (`user_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户资产总览表';

我准备做个资产管理系统，功能如下：
1.可以导入自己的月收入。
	1.设置月收入xxx
	2.设置发薪日时间，到达发薪日自动计算流水。（月、日）
2.可以导入自己的现有余额。
3.可以导入自己在某一天的突发收入（例如奖金、炒股等）。
4.可以导入自己的欠款，借贷类型，xx元，待还款xx月。还款日期。
5.可以导入固定消费类型，房租、房贷、饮食，固定消费类型也可以指定期限。
6.可以导入自己的存款计划和存款利率，如当前存款产品利率，计划每月存入多少，当前已有多少。
7.可以导入突发一次性欠款，可以分多次慢慢去删减这个欠款。
8.要有总览页面可以展示，要求可以很只管的看出自己的资产变化来进行管理，可以看出来自己的资产变化趋势，比如总资产，存款，剩余款，负债等。
9.有一键计算功能，在当前情况下，如果要还清所有负债需要多久，当前情况下，实现xx需要多久。

备注：
1.用户固定收入、支出（可以设置多个薪水和发薪日，这种就是相当于固定收入），可以设置触发时间（按周、按月、按年、按日、一周的某几日），（有状态，可以启动、禁用）（可以删除）
2.用户临时收入、支出
3.用户流水

  5. asset_recurring_transaction - 固定收支配置表

  CREATE TABLE `asset_recurring_transaction` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `transaction_id` VARCHAR(32) NOT NULL COMMENT '交易配置ID',
    `transaction_type` VARCHAR(32) NOT NULL COMMENT '交易类型; INCOME-收入、EXPENSE-支出',
    `transaction_name` VARCHAR(128) NOT NULL COMMENT '交易名称(如:月薪、房租)',
    `amount` DECIMAL(12,2) NOT NULL COMMENT '金额',
    `trigger_type` VARCHAR(32) NOT NULL COMMENT '触发类型; DAILY-按日、WEEKLY-按周、MONTHLY-按月、YEARLY-按年、CUSTOM_DAYS-自定义星期几',
    `trigger_value` VARCHAR(128) DEFAULT NULL COMMENT '触发值; 如按月则填1-31,按周则填MON,TUE,WED,按自定义则填1,3,5',
    `trigger_time` TIME DEFAULT NULL COMMENT '触发时间(如08:00:00)',
    `status` VARCHAR(32) NOT NULL DEFAULT 'ACTIVE' COMMENT '状态; ACTIVE-启用、DISABLED-禁用',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_transaction_id` (`transaction_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='固定收支配置表';


  6. asset_temporary_transaction - 临时收支记录表

  CREATE TABLE `asset_temporary_transaction` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `transaction_id` VARCHAR(32) NOT NULL COMMENT '交易ID',
    `transaction_type` VARCHAR(32) NOT NULL COMMENT '交易类型; INCOME-收入、EXPENSE-支出',
    `transaction_name` VARCHAR(128) NOT NULL COMMENT '交易名称(如:奖金、买手机)',
    `amount` DECIMAL(12,2) NOT NULL COMMENT '金额',
    `transaction_date` DATE NOT NULL COMMENT '交易日期',
    `transaction_time` TIME DEFAULT NULL COMMENT '交易时间',
    `category` VARCHAR(64) DEFAULT NULL COMMENT '分类(如:奖金、股票、购物、餐饮)',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_transaction_id` (`transaction_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='临时收支记录表';

  字段说明:
  - transaction_type: INCOME(收入)或EXPENSE(支出)
  - category: 可选,用于分类统计(如奖金、股票、购物、餐饮等)

  3. asset_transaction_flow - 资产流水表

  CREATE TABLE `asset_transaction_flow` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '自增ID',
    `user_id` VARCHAR(32) NOT NULL COMMENT '用户ID',
    `flow_id` VARCHAR(32) NOT NULL COMMENT '流水ID',
    `transaction_type` VARCHAR(32) NOT NULL COMMENT '交易类型; INCOME-收入、EXPENSE-支出',
    `transaction_name` VARCHAR(128) NOT NULL COMMENT '交易名称',
    `amount` DECIMAL(12,2) NOT NULL COMMENT '金额',
    `balance_before` DECIMAL(12,2) NOT NULL COMMENT '交易前余额',
    `balance_after` DECIMAL(12,2) NOT NULL COMMENT '交易后余额',
    `transaction_date` DATE NOT NULL COMMENT '交易日期',
    `transaction_time` DATETIME NOT NULL COMMENT '交易时间',
    `source_type` VARCHAR(32) NOT NULL COMMENT '来源类型; RECURRING-固定收支、TEMPORARY-临时收支',
    `source_id` VARCHAR(32) DEFAULT NULL COMMENT '来源ID(关联recurring_transaction或temporary_transaction)',
    `category` VARCHAR(64) DEFAULT NULL COMMENT '分类',
    `description` VARCHAR(255) DEFAULT NULL COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_flow_id` (`flow_id`)
  ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='资产流水表';
