 public class DynamicContext {

      /**
       * 节点放行标识
       * true = 继续执行下一个节点
       * false = 中断整个链路执行
       */
      private boolean proceed;

      /**
       * 跳过当前节点
       * true = 跳过当前节点的apply()方法，直接进入下一个节点
       * false = 正常执行当前节点
       */
      private boolean jump;

      /**
       * 默认构造函数
       * 默认状态：继续执行，不跳过
       */
      public DynamicContext() {
          this.proceed = true;
          this.jump = false;
      }

      /**
       * 存储共享数据的Map
       * key-value形式存储任意类型数据
       */
      private final Map<String, Object> dataObjects = new HashMap<>();

      /**
       * 向上下文中设置值
       * @param key 键
       * @param value 值
       */
      public <T> void setValue(String key, T value) {
          dataObjects.put(key, value);
      }

      /**
       * 从上下文中获取值
       * @param key 键
       * @return 值
       */
      public <T> T getValue(String key) {
          return (T) dataObjects.get(key);
      }

      public boolean isProceed() {
          return proceed;
      }

      public void setProceed(boolean proceed) {
          this.proceed = proceed;
      }

      public boolean isJump() {
          return jump;
      }

      public void setJump(boolean jump) {
          this.jump = jump;
      }
  }

ILogicHandler
~~~
  public interface ILogicHandler<T, D extends DynamicContext, R> {

      /**
       * 继续执行下一个节点
       * 设置 jump=false, proceed=true
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @return null
       */
      default R next(T requestParameter, D dynamicContext) {
          dynamicContext.setJump(false);
          dynamicContext.setProceed(true);
          return null;
      }

      /**
       * 停止执行链路
       * 设置 jump=false, proceed=false，中断整个链路
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @param result 返回结果
       * @return 结果
       */
      default R stop(T requestParameter, D dynamicContext, R result) {
          dynamicContext.setJump(false);
          dynamicContext.setProceed(false);
          return result;
      }

      /**
       * 跳过当前节点
       * 设置 jump=true, proceed=true
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @param result 返回结果
       * @return 结果
       */
      default R jump(T requestParameter, D dynamicContext, R result) {
          dynamicContext.setJump(true);
          dynamicContext.setProceed(true);
          return result;
      }

      /**
       * 主处理逻辑（必须实现）
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @return 处理结果
       * @throws Exception 处理异常
       */
      R apply(T requestParameter, D dynamicContext) throws Exception;

      /**
       * 前置处理（可选实现）
       * 在主逻辑执行前调用
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @return 前置处理结果
       * @throws Exception 处理异常
       */
      default R applyBefore(T requestParameter, D dynamicContext) throws Exception {
          dynamicContext.setJump(false);
          return null;
      }

      /**
       * 后置处理（可选实现）
       * 在主逻辑执行后调用（正常结束）
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @param result 主逻辑的返回结果
       * @throws Exception 处理异常
       */
      default void applyAfter(T requestParameter, D dynamicContext, R result) throws Exception {
      }

      /**
       * 异常处理（可选实现）
       * 当主逻辑抛出异常时调用
       * @param requestParameter 请求参数
       * @param dynamicContext 动态上下文
       * @param e 异常对象
       * @throws Exception 处理异常
       */
      default void applyAfterException(T requestParameter, D dynamicContext, Exception e) throws Exception {
      }
  }

public class SimplifiedBusinessLinkedList<T, D extends DynamicContext, R>
          extends LinkedList<ILogicHandler<T, D, R>>  // 直接继承
          implements ILogicHandler<T, D, R> {

      private final String name;

      public SimplifiedBusinessLinkedList(String name) {
          this.name = name;
      }

      public String getName() {
          return name;
      }

      @Override
      public R apply(T requestParameter, D dynamicContext) throws Exception {
          // 可以直接用增强 for 循环（因为继承了 LinkedList，实现了 Iterable）
          for (ILogicHandler<T, D, R> handler : this) {
              try {
                  R applyBefore = handler.applyBefore(requestParameter, dynamicContext);
                  if (!dynamicContext.isProceed()) {
                      handler.applyAfter(requestParameter, dynamicContext, applyBefore);
                      return applyBefore;
                  }

                  if (dynamicContext.isJump()) {
                      continue;
                  }

                  R apply = handler.apply(requestParameter, dynamicContext);
                  if (!dynamicContext.isProceed()) {
                      handler.applyAfter(requestParameter, dynamicContext, apply);
                      return apply;
                  }

              } catch (Exception e) {
                  handler.applyAfterException(requestParameter, dynamicContext, e);
                  throw e;
              }
          }
          throw new RuntimeException("责任链执行完毕，但没有任何处理器返回结果");
      }
  }

 ~~~
   public class LinkArmory<T, D extends DynamicContext, R> {

      /**
       * 业务链路实例
       * 持有配置好的责任链
       */
      private final BusinessLinkedList<T, D, R> logicLink;

      /**
       * 构造函数
       * 使用可变参数接收多个处理器并自动构建链路
       *
       * @param linkName 链路名称
       * @param logicHandlers 处理器数组（可变参数）
       */
      @SafeVarargs
      public LinkArmory(String linkName, ILogicHandler<T, D, R>... logicHandlers) {
          // 创建业务链路实例
          logicLink = new BusinessLinkedList<>(linkName);

          // 将所有处理器添加到链路中
          for (ILogicHandler<T, D, R> logicHandler : logicHandlers) {
              logicLink.add(logicHandler);
          }
      }

      /**
       * 获取配置好的业务链路
       * @return 业务链路实例
       */
      public BusinessLinkedList<T, D, R> getLogicLink() {
          return logicLink;
      }
  }
