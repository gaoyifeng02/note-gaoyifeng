  2. 泛型方法的设计

  // setValue的<T>是什么意思？
  public <T> void setValue(String key, T value)

  // 使用时：
  context.setValue("user", new UserInfo());  // T自动推导为UserInfo
  context.setValue("count", 100);            // T自动推导为Integer
  context.setValue("price", 99.9);           // T自动推导为Double

  // getValue的<T>避免强制类型转换
  public <T> T getValue(String key)

  // 使用时：
  UserInfo user = context.getValue("user");  // 直接返回UserInfo，不用强转
  Integer count = context.getValue("count"); // 直接返回Integer
